    class Solution:
        def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
            if not nums:
                return False
            package={}
            index=0
            for i in nums:
                if i not in package:
                    package[i]=index
                elif abs(package[i]-index)<=k:
                    return True
                else:
                    package[i]=index
                index+=1
            return False
