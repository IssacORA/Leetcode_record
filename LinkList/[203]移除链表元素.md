    class Solution:
        def removeElements(self, head: ListNode, val: int) -> ListNode:
            if not head:
                return
            output=head
            while output and output.val==val :
                output=output.next
            if not output:
                return

            h=output
            while h.next:
                if h.next.val ==val:
                    h.next=  h.next.next
                else:
                    h=h.next
            return output
