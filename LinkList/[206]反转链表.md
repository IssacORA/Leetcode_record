class Solution:
    def reverseList(self, head: ListNode) -> ListNode:
        stack=[None]
        h=head

        while h:
            temp_next=stack.pop()
            h0=h
            h=h.next
            h0.next=temp_next
            stack.append(h0)
        return stack[0]
