          def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> ListNode:
              if not headA or not headB:
                  return
              h1=headA
              h2=headB
              count=0
              while h1!=h2:
                  if count>=2 and (not h1.next or not h2.next):
                      return
                  if h1.next:
                      h1=h1.next
                  else:
                      h1 = headB
                      count += 1
                  if h2.next:
                      h2=h2.next
                  else:
                      h2 = headA
                      count += 1
              return h1
